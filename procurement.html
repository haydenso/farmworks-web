      function parseCsvFallback(csv){

        const rows=[];let row=[],field="",inQ=false;

        for(let i=0;i<csv.length;i++){const c=csv[i],n=csv[i+1];

          if(inQ){if(c==='"'){if(n==='"'){field+='"';i++;}else inQ=false;}else field+=c;}

          else{if(c==='"')inQ=true;else if(c===','){row.push(field);field="";}

          else if(c==='\n'||c==='\r'){if(c==='\r'&&n==='\n')i++;row.push(field);rows.push(row);row=[];field="";}

          else field+=c;}}

        if(field.length||row.length){row.push(field);rows.push(row);}

        if(!rows.length)return[];

        const hdrs=rows.shift().map(h=>String(h||"").replace(/^\uFEFF/,"").trim());

        return rows.filter(v=>v.some(c=>String(c||"").trim()!=="")).map(vals=>{const e={};hdrs.forEach((h,i)=>{e[h]=vals[i]!==undefined?vals[i]:"";});return e;});

      }
